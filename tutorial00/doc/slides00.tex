\begin{frame}
\frametitle{Trends in Computer Architecture}
\begin{itemize}
\item Power wall
\begin{itemize}
\item Power consumption is the limiting factor for exascale computing!
\item Tianhe 2: 33,8 PF (Linpack), 17.8 MW $\Rightarrow$ 500 MW
\item Clock rate stagnates but Moore's law is still valid
\end{itemize}
\item Memory wall
\begin{itemize}
\item Bandwidth not sufficient to sustain anywhere near peak performance
\item (Still) significant latency for main memory access 
\item Memory access is energetically costly
\item Memory is hierarchical: register, cache, local, remote, \ldots
\end{itemize}
\item ILP wall
\begin{itemize}
\item Automatic extraction of instruction level parallelism stagnates
\item Revival of vectorization in form of SIMD instructions
\end{itemize}
\item Other
\begin{itemize}
\item Heterogeneous node architectures: CPU + coprocessors
\item Reduced reliability due to immense number of components?
\end{itemize}
\end{itemize}
\begin{center}
\textbf{EXA-SCALE $\Rightarrow$ $10^6$ cores $\times$ $10^3$ threads $\times$ $10^9$ GFLOPS}
\end{center}
\end{frame}
